                    public class GetVideoCoursesListQueryHandler
                    : IRequestHandler<GetVideoCourseListQuery, List<VideoCourseViewModel>>
                    {
                        private readonly IAsyncRepository<VideoCourse> _videocourseRepository;
                        private readonly IMapper _mapper;
                    
                        public GetVideoCoursesListQueryHandler
                            (IAsyncRepository<VideoCourse> videocourseRepository,
                            IMapper mapper)
                        {
                            _mapper = mapper;
                            _videocourseRepository = videocourseRepository;
                        }
                    
                        public async Task<List<VideoCourseViewModel>>
                            Handle(GetVideoCourseListQuery request,
                            CancellationToken cancellationToken)
                        {
                            var allcourses = (await _videocourseRepository.ListAllAsync());
                            var allcoursesOrdered = allcourses.OrderBy(c => c.DateToBuy);
                    
                            return _mapper.Map<List<VideoCourseViewModel>>(allcoursesOrdered);
                        }
                    }
